import java.util.*;
import java.io.*;

public class Main {
	public static void main(String[] args) {
		Scanner scn = new Scanner(System.in);
		int n = scn.nextInt();
		int p = scn.nextInt();
		System.out.print(pow(n,p));
	}
	static int pow(int n, int p){
		if (p==1) return n;
		return pow(n,p-1)*n;
	}
}
