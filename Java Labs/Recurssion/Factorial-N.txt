import java.util.*;
import java.io.*;
public class Main {
	public static void main(String[] args) {
		Scanner scn = new Scanner(System.in);
		int n = scn.nextInt();
		System.out.print(factorial(n));
	}
	public static int factorial(int n ){
		if (n==1){
			return 1;
		}
		int temp = factorial(n-1);
		return temp*n;
	}
}
