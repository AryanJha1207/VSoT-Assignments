import java.util.*;
import java.io.*;

public class Main {
	public static void main(String[] args) {
		Scanner scn = new Scanner(System.in);
		int n = scn.nextInt();
		int sum = 0;
		System.out.print(transfer(n,sum));
	}
	public static int transfer(int n, int sum){
		if (n==0){
			return 0;
		}
		sum=1+transfer((n/10),sum);
		return sum;
	}
}
